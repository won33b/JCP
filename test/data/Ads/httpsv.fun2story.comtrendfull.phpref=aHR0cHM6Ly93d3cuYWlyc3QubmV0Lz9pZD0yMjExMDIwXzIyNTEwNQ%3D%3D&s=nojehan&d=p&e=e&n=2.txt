<script>
function getCookie(name) {
  var Found = false
  var start, end
  var i = 0

  while(i <= document.cookie.length) {
	start = i
	end = start + name.length
	if(document.cookie.substring(start, end) == name) {
	  Found = true
	  break 
	}
	i++
  }
 
  if(Found == true) {
	start = end + 1
	end = document.cookie.indexOf(";", start)
	  if(end < start)
		end = document.cookie.length
	return document.cookie.substring(start, end)
  }
  return ""
} 

function setCookie(name, value, tt) {
	var loadDt = new Date(); 
	var todayDate = new Date(Date.parse(loadDt) + 1000 * 60 * 60 * tt);
	document.cookie = name + "=" + escape( value ) + "; path=/; expires=" +   todayDate.toGMTString() + ";";
}

function popUnder(name, tt) {
	var noticeCookie=getCookie(name);  // 쿠키변수 이름
	if (noticeCookie != "no"){
		setCookie(name, "no", tt);
	}
}

function openToast(tt) {
	var noticeCookie=getCookie("toast");  // 쿠키변수 이름
	if (noticeCookie != "no"){
		var x = document.getElementById("toastopen");
		x.className = "open";
		setTimeout(function(){ x.className = x.className.replace("open", ""); }, 600000); // 600초(10분)
		setCookie("toast", "no", tt);
	}
}
