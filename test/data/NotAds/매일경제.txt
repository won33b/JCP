                    </div>   청소년의 심야 게임을 금지하는 이른바 '셧다운 제도'가 도입 10년 만에 폐지된다.
<br><br>국회는 11일 본회의에서 16세 미만의 청소년에게 오전 0시부터 오전 6시까지 인터넷 게임 제공을 제한하는 규정을 삭제하는 내용을 골자로 하는 청소년 보호법 개정안을 의결했다. 
<br><br>이 법안에는 인터넷게임 중독·과몰입 피해 청소년뿐만 아니라 피해 청소년 가족에게도 상담·교육 및 치료와 재활서비스를 지원할 수 있도록 하는 내용도 포함됐다. 
<br><div style="text-align:center; border:0; width:100%; margin:0 auto;"><div id="google_dfp_MC_2x1,fluid" style="float:inherit; padding: 20px 40px 9px 40px;"><script type="text/javascript">googletag.display("google_dfp_MC_2x1,fluid");</script></div></div><br>법 개정은 올해 8월 관계부처 합동으로 마련한 '셧다운제 폐지 및 청소년의 건강한 게임이용 환경 조성 방안'의 후속 조치로 이뤄졌다.
<br><br>당초 게임 셧다운제는 2000년대 초반 게임 과몰입이 사회적 문제로 떠오름에 따라 논의가 시작됐다.
<br><br>2005년 국회에서 청소년 보호법 개정안이 발의됐고, 2011년부터 시행됐다.
<br><br>이후 19·20대 국회에서 셧다운제의 합리적 운영을 위해 2차례 제도 개선을 추진했으나 법률 개정에는 이르지 못했다.

function submit_check(){

		cm.tomail.value = '';
		cm.tomail.focus();
		return;
	}

	if( (cm.tomail.value).match(/@/g).length > 1 ) {
		alert('받는 분의 메일은 두 개 이상일 수 없습니다. ');
		cm.tomail.value = '';
		cm.tomail.focus();
		return;
	}
	

	if( cm.frommail.value.length < 5 || CheckEmail( cm.frommail.value ) != true  ) {
		alert('보내는 분의 메일 주소를 정확히 입력해 주시기 바랍니다.');
		cm.frommail.value = '';
		cm.frommail.focus();
		return;
	}
	
	if( cm.subject.value != '' ) {
		if( str_count( cm.subject.value ) > 80 ) {
			alert( '제목은 80자까지만 작성이 가능합니다.' );
			var cut_memo = str_cut( cm.subject.value, 80, '' );
			cm.subject.value = cut_memo;		
			return;
		}	
	}

	if( cm.send_message.value != '' ) {
		if( str_count( cm.send_message.value ) > 200 ) {
			alert( '전할 메시지는 200자까지만 작성이 가능합니다.' );
			var cut_memo = str_cut( cm.send_message.value, 200, '' );
			cm.send_message.value = cut_memo;		
			return;
		}	
	}

	if( cm.fromname.value.length < 2 ) {
		alert('보내는 사람을 적어 주시기 바랍니다.');
		cm.fromname.focus();
		return;
	}

	$("#mail_mode").val("real");
	closeMail();
	cm.submit();
}

function CheckEmail( strText ) {
	if( strText.match( /^[a-zA-Z0-9_-]+@[a-zA-Z0-9_-]+.[a-zA-Z.]+$/ ) ) {
		return true;
	}
	else {
		return false;
	}
}

function str_count(expression) {
	var VLength=0;
	var temp = expression;
	var EscTemp;
	if(temp!='' & temp !=null) {
		for(var i=0;i<temp.length;i++) {
			if (temp.charAt(i)!=escape(temp.charAt(i))) {
				EscTemp=escape(temp.charAt(i));
				if (EscTemp.length>=6) VLength+=2;
				else VLength+=1;
			}
			else VLength+=1;
		}
	}
	return VLength;
}

function str_cut( str, len, tail ) {
    var l = 0;
	var ll = len;

    for( var i=0; i<str.length; i++ ) {
        l += ( str.charCodeAt(i) > 128 ) ? 2 : 1;
        if( l > ll ) return str.substring( 0, i ) + tail;
    }
    return str;
}
