function articleZoomin(cId) {
    var currSz = parseFloat($("#"+cId).css('font-size'));

    if(currSz>=24){
        alert("최대 폰트사이즈입니다");return;
    }

    Math.round(currSz);
    currSz++;
    $("#"+cId).css('font-size',currSz+'px');
    //$("#"+cId).css('line-height',(currSz+5)+'px');
}

// 글자작게
function articleZoomout(cId) {
    var currSz = parseFloat($("#"+cId).css('font-size'));

    if(currSz<=13){
        alert("최소 폰트사이즈입니다");return;
    }

    Math.round(currSz);
    currSz--;
    $("#"+cId).css('font-size',currSz+'px');
    //$("#"+cId).css('line-height',(currSz+5)+'px');
}

                                              <ul class="sub_title">
                                                  <li>이재명, ‘전두환씨’ 지칭…윤석열, 조문 의사 밝혔다가 철회</li>
                                                  <li>청와대 “공식 추모 메시지 아냐…전두환 사망 관련 브리핑”</li>

<div class="imgBox ib_center" id="imgBox_2051267">
<div id="captionBox" style="width:640px; clear:both; float:none;"><img height="359" id="imgs_2051267" photo_no="2051267" src="https://image.ajunews.com/content/image/2021/11/23/20211123170046946019.jpg" width="640" />
<p class="p_caption" style="text-align:left;">이재명 더불어민주당 대선 후보가 23일 오전 서울 여의도 중앙당사에서 &#39;디지털 대전환&#39; 공약을 발표하고 있다. [사진=연합뉴스]</p>
</div>
</div>

<div class="imgBox ib_center" id="imgBox_2051269">
<div id="captionBox" style="width:640px; clear:both; float:none;"><img height="426" id="imgs_2051269" photo_no="2051269" src="https://image.ajunews.com/content/image/2021/11/23/20211123170117381240.jpg" width="640" />
<p class="p_caption" style="text-align:left;">윤석열 국민의힘&nbsp;대선 후보가 23일 오전 서울 영등포 국민의힘 서울시당에서 열린 서울시당 핵심당직자 화상 전체회의에 참석해 인사말을 하고 있다. [사진=연합뉴ㅠ스]</p>
</div>
</div>
정치권은 23일 전두환 전 대통령의 사망과 관련해 입장이 극명하게 갈렸다. 청와대는 조화를 보내지 않고 조문도 하지 않겠다는 뜻을 밝혔다.<br />
&nbsp;<br />
박경미 청와대 대변인은 이날 춘추관 브리핑을 통해 &ldquo;전 전 대통령의 명복을 빌고, 유가족에게 위로의 말씀을 전한다&rdquo;면서 &ldquo;끝내 역사의 진실을 밝히지 않고 진정성 사과 없는 데 유감&rdquo;이라고 전했다. 그러면서 &ldquo;청와대 차원의 조화나 조문 계획은 없다&rdquo;고 덧붙였다.<br id="br_2" />
&nbsp;<br id="br_3" />
청와대 핵심 관계자는 &lsquo;전 전 대통령에 대한 공식 추모 메시지냐&rsquo;는 기자들의 질문에도 &ldquo;추모 메시지라기보다는 브리핑 제목은 &lsquo;전두환 사망 관련 대변인 브리핑&rsquo;&rdquo;이라고 의미를 축소했다.<br id="br_4" />
&nbsp;<br id="br_5" />
더불어민주당을 비롯한 여권은 애도보다 전 전 대통령의 과거에 대한 비판의 목소리를 앞세웠다.<br id="br_6" />
&nbsp;<br id="br_7" /><div class="banner_wrap banner_wrap_01" id="banner20211104"></div><div class="relate_box" sty><strong>관련기사</strong><ul><li><a href="/view/20211123173056584">장세동·허화평·허삼수·허문도…실세의 그늘</a></li><li><a href="/view/20211123133016258">이재명 &quot;조문 생각 안 해&quot; 윤석열 &quot;전직 대통령이니 가야&quot;</a></li></ul></div>
반면 야당은 과오를 인정하면서도 &ldquo;안타깝다&rdquo;는 쪽에 무게를 뒀다. 대선주자들은 일제히 조문을 하지 않겠다면서도 입장은 극명하게 엇갈렸다.<br id="br_8" />
&nbsp;<br id="br_9" />
이재명 민주당 대선 후보는 이날 첫 대선 공약 발표 후 기자들과 만나 &ldquo;전두환씨는 명백하게 확인된 것처럼 내란 학살 사건 주범&rdquo;이라며 &ldquo;이 중대 범죄 행위를 인정하지도 않았다. 참으로 아쉽게 생각하고, 다시는 이런 일이 생기지 않았으면 좋겠다&rdquo;고 밝혔다.<br id="br_10" />
&nbsp;<br id="br_11" />
이 후보는 &ldquo;최하 수백명의 사람을 살상했던, 자신의 사적 욕망을 위해서 국가권력을 찬탈했던, 결코 용서받을 수 없는 범죄에 대해 마지막 순간까지도 국민께 반성하고 사과하지 않았다&rdquo;면서 &ldquo;아직도 여전히 미완 상태인 광주 5&middot;18 민주화운동의 진상이 드러날 수 있도록, 당시 사건 관련자들의 양심선언을 기대한다&rdquo;고 말했다.<br id="br_12" />
&nbsp;<br id="br_13" />
이 후보는 전 전 대통령을 &lsquo;전두환씨&rsquo;라고 지칭했다. 조문 계획에 대해선 &ldquo;현재 상태로는 아직 조문 생각을 하지 않고 있다&rdquo;고 했다.<br id="br_14" />
&nbsp;<br id="br_15" />
민주당 역시 노태우 전 대통령의 명암을 동시에 조명한 것과 달리 전 전 대통령이 과거에 대해 참회하지 않은 점을 지적하며 아쉬움을 표했다.<br id="br_16" />
&nbsp;<br id="br_17" />
고용진 선대위 수석대변인은 이날 브리핑에서 전 전 대통령에 대해 &ldquo;고인의 명복을 빈다&rdquo;면서도 &ldquo;아쉽게도 고인은 진정한 사과와 참회를 거부하고 떠났다. 참으로 아쉽다&rdquo;고 밝혔다.<br id="br_18" />
&nbsp;<br id="br_19" />
앞서 배포한 서면 브리핑 자료에는 &lsquo;애도를 표한다&rsquo;는 표현이 있었지만 실제 브리핑 현장에선 이를 삭제했다.<br id="br_20" />
&nbsp;<br id="br_21" />
심상정 정의당 대선 후보는 &ldquo;역사를 인식한다면 국가장 얘기는 감히 입에 올리지 않기를 바란다&rdquo;면서 &ldquo;성찰 없는 죽음은 그조차 유죄&rdquo;라고 목소리를 높였다.<br id="br_22" />
&nbsp;<br id="br_23" />
국민의힘 측은 말을 아꼈다. 윤석열 국민의힘 대선 후보는 이날 대선 경선 예비후보들과의 오찬에 앞서 전 전 대통령 사망 소식에 대해 &ldquo;삼가 조의를 표하고 유족에게 위로의 말씀을 드린다&rdquo;고 말했다.<br id="br_24" />
&nbsp;<br id="br_25" />
윤 후보는 &lsquo;전 전 대통령이 5&middot;18 관련 별다른 사과도 하지 않고 세상을 떴다&rsquo;는 기자들의 질문에 &ldquo;돌아가셨고 상중이니까 정치적인 얘기를 그분하고 관련지어서 얘기하는 건 시의적절하지 않은 거 같다&rdquo;고 즉답을 피했다.<br id="br_26" />
&nbsp;<br id="br_27" />
윤 후보는 조문 의사를 밝혔다가 철회를 하기도 했다. 앞서 이준석 국민의힘 대표는 이날 자신의 페이스북에 &ldquo;당을 대표해 조화는 보내도록 하겠다&rdquo;면서도 &quot;전 전 대통령 상가에 따로 조문할 계획이 없다&ldquo;고 밝혔다.<br id="br_28" />
&nbsp;<br id="br_29" />
다만 이 대표는 &ldquo;당내 구성원들은 고인과의 인연이나 개인적 판단에 따라 자유롭게 조문 여부를 결정하셔도 된다&rdquo;고 말했다.&nbsp;</div>

/* 로그인 */
function doLogin(type) {
    $.ajax({
        url: _SOCIAL_URI_AUTH_,
        data: {'action':'login', 'type':type, 'rdUrl':$('#sr_rd_url').val()},
        success: function(data) {
            var json = jQuery.parseJSON(data);
            var loginUrl = json.loginUrl;
            $(location).attr('href', loginUrl);
        } 
    });
}

/* 로그아웃 */
function doLogout() {
    $.ajax({
        url: _SOCIAL_URI_AUTH_,
        data: {'action':'logout'},
        success: function(data) {
            var json = jQuery.parseJSON(data);
            if(json.status=='success') { 
                location.reload();
            }
        }
