				element.innerHTML = this.getAdHTML(size);
				element.classList.remove('vm-placement');
			};

			this.showAdSkyscraper = function(){
				if (this.getMobileOS() !== "unknown") {
					return;
				}

				var elementLeft = document.createElement("div");
				elementLeft.innerHTML = this.getAdHTML("skyscraper");
				elementLeft.setAttribute('style', 'position: absolute; left: 50%; top: 106px; margin-left: -725px;');
				document.body.appendChild(elementLeft);

				var elementRight = document.createElement("div");
				elementRight.innerHTML = this.getAdHTML("skyscraper");
				elementRight.setAttribute('style', 'position: absolute; right: 50%; top: 106px; margin-right: -725px;');
				document.body.appendChild(elementRight);

				$(elementRight).scrollToFixed({
					marginTop: 60
				});
			};

			this.getAdHTML = function(size){
				var lang = (this.getLang() === 'ko_KR' ? 'ko' : 'en');
				return "<a href=\"" + this.getAdLink(this.adIndex) + "\" target=\"_blank\" class=\"lifeOnwer-" + randAdIdx + "\">" +
						"<img src='//opgg-static.akamaized.net/images/lifeowner/opgg/" + lang + "/" + adIndex + "-" + size + ".png'>" +
						"</a>";
			};
			this.getAdLink = function(adIndex){
				var linkKey = this.getMobileOS() + '-' + adIndex;
				switch (linkKey) {
					case "Android-1":
						return "market://details?id=gg.op.lol.android&referrer=utm_source%3Dopgg%26utm_medium%3Dbanner01%26utm_content%3Dapp";
					case "Android-2":
						return "market://details?id=gg.op.lol.android&referrer=utm_source%3Dopgg%26utm_medium%3Dbanner02%26utm_content%3Dlol";
					case "Android-3":
						return "market://details?id=gg.op.lol.android&referrer=utm_source%3Dopgg%26utm_medium%3banner03%26utm_content%3Dpubg";
					case "unknown-1":
						return "https://play.google.com/store/apps/details?id=gg.op.lol.android&referrer=utm_source%3Dopgg%26utm_medium%3Dbanner01%26utm_content%3Dapp";
					case "unknown-2":
						return "https://play.google.com/store/apps/details?id=gg.op.lol.android&referrer=utm_source%3Dopgg%26utm_medium%3Dbanner02%26utm_content%3Dlol";
					case "unknown-3":
						return "https://play.google.com/store/apps/details?id=gg.op.lol.android&referrer=utm_source%3Dopgg%26utm_medium%3Dbanner03%26utm_content%3Dpubg";
					default:
						return "itms-apps://itunes.apple.com/kr/app/apple-store/id605722237";
				}
			};
			this.getMobileOS = function(){
				var userAgent = navigator.userAgent || navigator.vendor;

				if (/android/i.test(userAgent)) {
					return "Android";
				}

				if (/iPad|iPhone|iPod/.test(userAgent)) {
					return "iOS";
				}

				return "unknown";
			};
			this.getLang = function(){
				return document.documentElement.lang || "en";
			};

			this.slots = [
				{
					'id': '6076ad1ba1ede816581526af',
					'size': '970x90'
				},
				{
					'id': '5a84640c46e0fb0001fc3bb0',
					'size': '300x250'
				},
				{
					'id': '6076ad3d9ada6778be32cb33',
					'size': 'skyscraper'
				},
			];
			this.getCookie = function(name){
				var match = document.cookie.match(new RegExp('(^| )' + name + '=([^;]+)'));
				if (match) return match[2];
			};
			this.bindClickEvent = function(){
				$(".lifeOnwer-" + randAdIdx).click(function(evt){
					$.OP.GG.tracker.combine.sendEvent(['FB', 'Pixel'], 'Opgg', 'ClickAd', {
						Location: location_,
						subLocation: subLocation,
						BannerType: adIndex
					});
				});
			};

			this.adIndex = Math.floor(Math.random() * (this.getMobileOS() === "iOS" ? 2 : 3)) + 1;
			this.location_ = "index";
			this.subLocation = "";


			slots.forEach(function(slot){
				if (self.getMobileOS() === "unknown" && slot.size === "300x250") {
					return;
				}
				var element = document.querySelector('.vm-placement[data-id="' + slot.id + '"]');
				if (element) {
					self.showAd(element, slot.size);
				}
			});
			this.bindClickEvent();
		};
