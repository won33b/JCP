<p><span style="FONT-SIZE: 12pt; COLOR: #0055ff"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 10pt; COLOR: #000000">기존의 Java 같은 경우는 int 변수, String 변수 등 해당하는 형으로 선언하는 것이 맞지만 자바스크립트는 조금 다릅니다. 일단 모든 변수를 var로 선언 가능합니다. </span></span></span></span></p>
<p><span style="FONT-SIZE: 12pt; COLOR: #0055ff"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 10pt; COLOR: #000000"></span></span></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt; COLOR: #0055ff"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 10pt; COLOR: #000000">- 문자형을 숫자형으로 변환하기</span></span></span></span></p>
<p><span style="FONT-SIZE: 12pt; COLOR: #0055ff"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 12pt; COLOR: #000000"><span style="FONT-SIZE: 10pt; COLOR: #000000"></span></span></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><strong><span style="COLOR: #0055ff">var</span></strong> 변수 = parseInt(문자);&nbsp;&nbsp;&nbsp;&nbsp;<span style="FONT-SIZE: 10pt">//문자를 정수형 숫자로 변환해줌</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="COLOR: #0055ff"><strong>var</strong></span> 변수 = parseFloat(문자);&nbsp;&nbsp;&nbsp;&nbsp;&nbsp;<span style="FONT-SIZE: 10pt">//문자를 실수형 숫자로 변환해줌</span><span style="FONT-SIZE: 10pt"></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="COLOR: #0055ff"><strong>var</strong></span> 변수 = Nember(문자);&nbsp;&nbsp;&nbsp;&nbsp;<span style="FONT-SIZE: 10pt">//문자를 정수&amp;실수형 숫자로 변환해줌</span><span style="FONT-SIZE: 10pt"></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">예제)</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var x = "999";&nbsp;&nbsp;&nbsp;&nbsp;//문자형 999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var&nbsp;y&nbsp;= "99.99";&nbsp;&nbsp;&nbsp;&nbsp;//문자형 99.99</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var&nbsp;a = parseInt(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 정수 999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var&nbsp;b = parseInt(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 정수 99</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = parseFloat(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 실수 999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = parseFloat(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 실수 99.99</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = Number(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 정수 999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = Number(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 실수 99.99</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var x = "a999";&nbsp;&nbsp;&nbsp;&nbsp;//문자형 a999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var y = "a99.99";&nbsp;&nbsp;&nbsp;&nbsp;//문자형 a99.99</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = parseInt(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = ParseInt(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = parseFloat(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = parseFloat(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = Number(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = Number(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"><strong><span style="COLOR: #ff0000">※ 문자열 맨앞에 문자로 시작하면 형변환을 해도 값을 인식하지 못한다.</span></strong></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var x = "999a9";&nbsp;&nbsp;&nbsp;&nbsp;//문자열 999a9</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var y = "99.9a9";&nbsp;&nbsp;&nbsp;&nbsp;//문자열 99.9a9</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = parseInt(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = parseInt(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 99</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = parseFloat(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 999</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = parseFloat(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 99.9</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = Number(x);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var b = Number(y);&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 NaN</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">- 숫자형을 문자형으로 변환하기</span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><strong><span style="COLOR: #0055ff">var</span></strong> 변수 = String(숫자);<span style="FONT-SIZE: 10pt; COLOR: #000000">&nbsp;&nbsp;&nbsp;&nbsp;//숫자를 문자로 변환해줌</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="COLOR: #0055ff"><strong>var</strong></span> 변수 = 숫자.toString(진법);&nbsp;&nbsp;&nbsp;&nbsp;<span style="FONT-SIZE: 10pt">//숫자를 문자로 변환해줌 - 변환하면서 진법을 바꿀 수 있음</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="COLOR: #0055ff"><strong>var</strong></span> 변수 = 숫자.toFixed(소수자리수);<span style="FONT-SIZE: 10pt">&nbsp;&nbsp;&nbsp;&nbsp;//숫자를 문자로 변환해줌 - 실수형의 소수점 자리를 지정할 수 있음</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="COLOR: #0055ff"><strong>var</strong></span> 변수 = 숫자 + "문자열";<span style="FONT-SIZE: 10pt">&nbsp;&nbsp;&nbsp;&nbsp;//숫자와 문자열을 한 문자열로 더해줌</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var x = 123;&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 123</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = String(x);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 123</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toString();&nbsp;&nbsp;&nbsp;&nbsp;//문자형 123</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toString(2);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 1111011(2진법)</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toString(16);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 7b(16진법)</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toFixed();&nbsp;&nbsp;&nbsp;&nbsp;//문자형 123</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toFixed(1);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 123.0</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var y = 99.999;&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 99.999</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toFixed();&nbsp;&nbsp;&nbsp;&nbsp;//문자형 99</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toFixed(1);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 99.9</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toFixed(3);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 99.999</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = x.toFixed(4);&nbsp;&nbsp;&nbsp;&nbsp;//문자형 99.9990</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">확률을 표현해줄 때 유용하다.</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt"></span></span></span>&nbsp;</p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var z = 9;&nbsp;&nbsp;&nbsp;&nbsp;//숫자형 9</span></span></span></p>
<p><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 12pt"><span style="FONT-SIZE: 10pt">var a = a + "ElNino Torres";&nbsp;&nbsp;&nbsp;&nbsp;//문자형 9ElNino Torres</span></span></span></p>
