    function refreshPreference(myPreference, jsonPreference, cid) {
    
        if(jsonPreference) {
    
            var elItemLikes = document.getElementById("item_likes-"+cid)
            var elItemHates = document.getElementById("item_hates-"+cid)
    
            elItemLikes.textContent = jsonPreference.likes
            elItemHates.textContent = jsonPreference.hates
    
            if(myPreference) {
    
                if(myPreference == 'like') {
    
                    // if you've already liked, cancel thumb up
                    if(elItemLikes.style.color == 'rgb(255, 152, 33)') {
    
                        elItemLikes.style.color = '#000000'
                    }
                    else {
    
                        elItemLikes.style.color = '#ff9821'
                        elItemHates.style.color = '#000000'
                    }
                }
                else if(myPreference == 'hate') {
    
                    // if you've already hated, cancel thumb down
                    if(elItemHates.style.color == 'rgb(0, 42, 221)') {
    
                        elItemHates.style.color = '#000000'
                    }
                    else {
    
                        elItemHates.style.color = '#002add'
                        elItemLikes.style.color = '#000000'
                    }
                }
                else {
    
                    console.log('invalid preference:', myPreference)
                }
            }
        }
    }
