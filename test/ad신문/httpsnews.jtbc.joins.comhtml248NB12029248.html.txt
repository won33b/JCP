        function fnRandom(num, range) {
            var k = 0, randomNum = [];
            while (k != num) {
                var ramdomNum = Math.floor(Math.random() * range);
                for (var j = 0; j < randomNum.length; j++) {
                    if (randomNum[j] == ramdomNum) break;
                }
                if (j == randomNum.length) { randomNum.push(ramdomNum); k++; }
            }
            return randomNum;
        }

        function fnHotNewsOpenClose() {
            var nStatus = $("#classStatus").val();
            if (nStatus % 2 == 0) { /* 0 : Close, 1: Open */
                $("#openCloseHotNews").removeClass();
                $("#openCloseHotNews").addClass("bd is_hide");
                $("#openCloseHotNews").next().hide(); /* Hide Arrow */
                $("#openCloseHotNews > a").html("HOT NEWS열기");
            }
            else {
                $("#openCloseHotNews").removeClass();
                $("#openCloseHotNews").addClass("bd is_show");
                if (nStatus == 3) /* 최초 닫힘 상태일 경우 Rolling 함수가 중복되므로 기본값 3으로 주어 fnHotNewsRollingEvent함수가 중복되지 않게 함 */
                    fnHotNewsRollingEvent("shownews", 0);

                $("#openCloseHotNews").next().show(); /* Show Arrow */
                $("#openCloseHotNews > a").html("HOT NEWS닫기");
            }
            nStatus++;

            if (nStatus >= 2) /* CheckValue is 0, 1 */
                nStatus = 0;

            $("#classStatus").val(nStatus);
        }
