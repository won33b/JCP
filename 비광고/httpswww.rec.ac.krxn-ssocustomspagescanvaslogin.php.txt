<script language="javascript" type="text/javascript">
    function getCookie(name) {
        var v = document.cookie.match('(^|;) ?' + name + '=([^;]*)(;|$)');
        return v ? v[2] : null;
    }

    function keyDown(e) {
        var intKey = e.keyCode ? e.keyCode : e.whichCode;
        if (intKey == 13) { //enter 키 입력시 로그인 가능
            OnLogon();
        }
    }

    function OnLogon() {
        var luiui = document.forms["form1"].login_user_id_user_input.value;
        if (luiui == "") {
            alert("사용자 ID를 입력하세요");
            return;
        }
        if (document.forms["form1"].pseudonym_session_password.value == "") {
            alert("사용자의 로그온 비밀번호를 입력하세요");
            return;
        }

        var upfx = document.getElementById('user-login-prefix').value;
        document.getElementById('pseudonym_session_unique_id').value = upfx + luiui;

        document.forms["form1"].action = "https://canvas.rec.ac.kr/login/canvas";
        document.forms["form1"].target = "_top";
        document.forms["form1"].submit();
    }
</script>
