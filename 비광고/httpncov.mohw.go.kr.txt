	function search( where ){
        var thisForm = $("#"+ where);
        var thisSearch = $(thisForm).find(".searchTerm");
        var cont = "";
        $(thisSearch).each(function() {
            cont += $(this).val();
        });

	//특수문자 체크
	if (containsChars(cont,"~!@#$%^&*()+|`=\\[]{};:'\".<>/?")) {
		alert("특수문자는 제외하고 검색해주시기 바랍니다.");
		$(thisSearch).val("");
		event.preventDefault();
		return false;
	}else if ( cont == "" || cont == null || cont == undefined){
		alert("검색어를 입력하세요");
		$(thisSearch).focus();
		event.preventDefault();
		return false;
	}
	
	return true;		
	
}
